# 驾驶陪练系统单元测试文档

## 一、单元测试概述
本单元测试文档针对驾驶陪练系统的核心功能模块进行测试，确保系统的各个功能组件能够正常工作。测试采用pytest框架进行后端单元测试，Vitest进行前端组件测试。

## 二、测试环境
- **后端测试框架**：pytest
- **前端测试框架**：Vitest
- **数据库**：SQLLite（测试环境）
- **Python版本**：3.10+
- **Node.js版本**：18+

#### 4.2.1 Login.vue测试
```javascript
import { describe, it, expect } from 'vitest'
import { mount } from '@vue/test-utils'
import Login from '@/components/Login.vue'

describe('Login.vue', () => {
  it('renders login form', () => {
    const wrapper = mount(Login)
    expect(wrapper.find('form').exists()).toBe(true)
    expect(wrapper.find('input[type="tel"]').exists()).toBe(true)
    expect(wrapper.find('input[type="password"]').exists()).toBe(true)
    expect(wrapper.find('button[type="submit"]').exists()).toBe(true)
  })

  it('validates phone number', () => {
    const wrapper = mount(Login)
    const phoneInput = wrapper.find('input[type="tel"]')
    phoneInput.setValue('123456789')
    wrapper.find('button[type="submit"]').trigger('click')
    expect(wrapper.find('.error-message').text()).toContain('请输入有效的手机号')
  })

  it('validates password length', () => {
    const wrapper = mount(Login)
    const phoneInput = wrapper.find('input[type="tel"]')
    const passwordInput = wrapper.find('input[type="password"]')
    phoneInput.setValue('13800138000')
    passwordInput.setValue('123')
    wrapper.find('button[type="submit"]').trigger('click')
    expect(wrapper.find('.error-message').text()).toContain('密码长度不能少于6位')
  })
})
```

```javascript
import { describe, it, expect } from 'vitest'
import { mount } from '@vue/test-utils'
import ScheduleList from '@/components/ScheduleList.vue'

describe('ScheduleList.vue', () => {
  it('renders schedule list', () => {
    const schedules = [
      { id: 1, coach_name: '王五', schedule_date: '2023-01-01', schedule_time: '上午', status: '待确认' },
      { id: 2, coach_name: '赵六', schedule_date: '2023-01-02', schedule_time: '下午', status: '已确认' }
    ]
    const wrapper = mount(ScheduleList, {
      props: { schedules }
    })
    expect(wrapper.findAll('.schedule-item')).toHaveLength(2)
  })

  it('triggers book event when book button is clicked', () => {
    const schedules = [
      { id: 1, coach_name: '王五', schedule_date: '2023-01-01', schedule_time: '上午', status: '待确认' }
    ]
    const wrapper = mount(ScheduleList, {
      props: { schedules }
    })
    wrapper.find('.book-button').trigger('click')
    expect(wrapper.emitted('book')).toBeTruthy()
    expect(wrapper.emitted('book')[0]).toEqual([1])
  })
})
```
